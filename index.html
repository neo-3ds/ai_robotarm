// --- 전역 변수 --- let device; let serialPort; let writer; let currentMode = 'gesture'; let isFullscreen = false; let angles = { ud: 90, lr: 90, gr: 90 }; let sendTimer = null; // [수정] USB 연결 중 카메라 연산을 잠시 멈추기 위한 플래그 변수 let isConnecting = false; } expandBtn.onclick = (e) => { e.stopPropagation(); isFullscreen = !isFullscreen; if (isFullscreen) viewportContainer.classList.add('is-fullscreen'); else viewportContainer.classList.remove('is-fullscreen'); setTimeout(() => { canvasElement.width = viewportContainer.offsetWidth; canvasElement.height = viewportContainer.offsetHeight; }, 100); }; // --- [연결 버그 및 권한 중복 요청 해결 로직] --- connectBtn.onclick = async () => { // [수정] 카메라 stop 대신 플래그를 true로 바꾸어 MediaPipe 연산만 일시 중단 isConnecting = true; if (dist < 0.075) angles.gr = 180; else if (dist > 0.125) angles.gr = 0; else angles.gr = 90; } for (const landmark of landmarks) { canvasCtx.beginPath(); canvasCtx.arc((1 - landmark.x) * canvasElement.width, landmark.y * canvasElement.height, 5, 0, 2 * Math.PI); } }, width: 640, height: 480 }); camera.start(); // --- 모바일 터치 제어 로직 --- viewportContainer.addEventListener('touchstart', handleTouch); viewportContainer.addEventListener('touchmove', handleTouch); viewportContainer.addEventListener('touchend', () => { document.getElementById('touchPoint').style.display = 'none'; }); function handleTouch(e) { if (currentMode !== 'touch' || e.target.closest('#expandBtn')) return; e.preventDefault(); const rect = viewportContainer.getBoundingClientRect(); const touch = e.touches[0]; document.getElementById('touchPoint').style.top = (touch.clientY - rect.top) + 'px'; if (x > 0.8) { angles.gr = mapAngle(y); } else { angles.lr = mapAngle(1 - x); angles.ud = mapAngle(y); } } // --- 데스크탑 마우스 커서 및 휠 제어 로직 --- viewportContainer.addEventListener('mousemove', handleMouseMove); const y = (e.clientY - rect.top) / rect.height; document.getElementById('touchPoint').style.display = 'block'; document.getElementById('touchPoint').style.left = (e.clientX - rect.left) + 'px'; document.getElementById('touchPoint').style.top = (e.clientY - rect.top) + 'px'; if (x > 0.8) { angles.gr = mapAngle(y); } else { angles.lr = mapAngle(1 - x); angles.ud = mapAngle(y); } } function handleMouseWheel(e) { if (currentMode !== 'touch') return; e.preventDefault(); if (e.deltaY < 0) { angles.gr = 0;
